        function parseURLPersonalization() {
            const urlParams = new URLSearchParams(window.location.search);
            const rawName = urlParams.get('name');
            if (rawName) {
                const sanitizedName = rawName.trim();
                // Select all dynamic title placeholder classes
                document.querySelectorAll('.visitor-name').forEach(el => {
                    el.textContent = sanitizedName;
                });
                // Update landing subtitle logic

            for (let i = 1; i < totalSlides; i++) {
                const barBg = document.createElement('div');
                barBg.className = 'h-1 flex-grow bg-zinc-800 rounded-full overflow-hidden relative';
                
                const barFill = document.createElement('div');
                barFill.id = `bar-fill-${i}`;
                barFill.className = 'h-full bg-white w-0 transition-all ease-linear';
                
                barBg.appendChild(barFill);
                container.appendChild(barBg);
            }
        }

        function startWrapped() {
            document.getElementById('progress-container').classList.remove('hidden');
            document.getElementById('nav-controls').classList.remove('hidden');
            goToSlide(1);
        }

        function goToSlide(targetIdx) {
            if (targetIdx < 0 || targetIdx >= totalSlides) return;

            // Remove active tags from existing context slide target
            document.getElementById(`slide-${currentSlideIdx}`).classList.remove('active');
            
            currentSlideIdx = targetIdx;
            
            const nextSlideEl = document.getElementById(`slide-${currentSlideIdx}`);
            nextSlideEl.classList.add('active');

            // Handle bottom menu tracking updates
            if (currentSlideIdx > 0) {
                document.getElementById('slide-index-indicator').textContent = `${currentSlideIdx}/${totalSlides - 1}`;
            }

            // Fire slide specific logic hooks (like counter tickers or celebration systems)
            handleSlideAnimations(currentSlideIdx);
            updateProgressBarsUI();
            startProgressBarTimer();
        }

        function nextSlide() {
            if (currentSlideIdx < totalSlides - 1) {
                goToSlide(currentSlideIdx + 1);
            } else {
                // If it reaches the ultimate slide layout frame loop context safely
                goToSlide(1);
            }
        }

        function prevSlide() {
            if (currentSlideIdx > 1) {
                goToSlide(currentSlideIdx - 1);
            } else if (currentSlideIdx === 1) {
                // Return to landing context frame completely
                document.getElementById('progress-container').classList.add('hidden');
                document.getElementById('nav-controls').classList.add('hidden');
                document.getElementById(`slide-${currentSlideIdx}`).classList.remove('active');
                currentSlideIdx = 0;
                document.getElementById('slide-0').classList.add('active');
                clearInterval(progressBarInterval);
            }
        }

        // Automatic playback state updates handling
        function startProgressBarTimer() {
            clearInterval(progressBarInterval);
            if (currentSlideIdx === 0) return;

            const activeBar = document.getElementById(`bar-fill-${currentSlideIdx}`);
            if (!activeBar) return;

            let startTime = performance.now();
            
            progressBarInterval = setInterval(() => {
                let elapsed = performance.now() - startTime;
                let progressPercent = Math.min((elapsed / SLIDE_DURATION) * 100, 100);
                
                activeBar.style.width = `${progressPercent}%`;

                if (elapsed >= SLIDE_DURATION) {
                    clearInterval(progressBarInterval);
                    nextSlide();
                }
            }, 30);
        }

        // Align status variables on skipped or backward context bar elements instantly
        function updateProgressBarsUI() {
            for (let i = 1; i < totalSlides; i++) {
                const fill = document.getElementById(`bar-fill-${i}`);
                if (!fill) continue;
                
                // Clear active processing transitions transitions to override instantly
                fill.style.transition = 'none'; 
                
                if (i < currentSlideIdx) {
                    fill.style.width = '100%';
                } else if (i === currentSlideIdx) {
                    fill.style.width = '0%';
                    // Re-apply matching sleek smooth animation transition rules context
                    setTimeout(() => { fill.style.transition = 'width 30ms linear'; }, 10);
                } else {
                    fill.style.width = '0%';
                }
            }
        }

        function handleSlideAnimations(slideIdx) {
            // SLIDE 1: Run numeric metrics ticker animations 
            if (slideIdx === 1) {
                const targets = document.querySelectorAll('#slide-1 [data-target]');
                targets.forEach(el => {
                    const maxVal = parseInt(el.getAttribute('data-target'), 10);
                    animateNumberValue(el, 0, maxVal, 1500);
                });
            }
            // SLIDE 6: Fire confetti automatically upon reaching the ultimate reveal layout slide
            if (slideIdx === 6) {
                triggerCelebrationConfetti();
            }
        }

        // Pure standard JS mathematical calculation ticker function
        function animateNumberValue(element, start, end, duration) {
            let startTimestamp = null;
            const step = (timestamp) => {
                if (!startTimestamp) startTimestamp = timestamp;
                const progress = Math.min((timestamp - startTimestamp) / duration, 1);
                element.innerText = Math.floor(progress * (end - start) + start).toLocaleString();
                if (progress < 1) {
                    window.requestAnimationFrame(step);
                }
            };
            window.requestAnimationFrame(step);
        }

        function triggerCelebrationConfetti() {
            const duration = 3 * 1000;
            const end = Date.now() + duration;

            (function frame() {
                confetti({
                    particleCount: 4,
                    angle: 60,
                    spread: 55,
                    origin: { x: 0 },
                    colors: ['#1DB954', '#ffffff', '#ff41b4']
                });
                confetti({
                    particleCount: 4,
                    angle: 120,
                    spread: 55,
                    origin: { x: 1 },
                    colors: ['#1DB954', '#ffffff', '#ff41b4']
                });

                if (Date.now() < end) {
                    requestAnimationFrame(frame);
                }
            }());
        }

        }
